283 void page_fault(registers_t regs){
284  uint32_t fault_addr = read_cr2();
285  int present = !(regs.err_code & 0x1); /* Page not present */
286  int rw = regs.err_code & 0x2; /* Page is read only */
287  int user = regs.err_code & 0x4; /* User mode */
288  int reserved = regs.err_code & 0x8; /* Reserved bits is writed */
289  int id = regs.err_code & 0x10; /* */
290  qemu_log("Page fault: ");
291  char* msg = "Переполнение памяти буфера";
292  if (present){
293  qemu_log("NOT PRESENT, ");
294  msg = "Память по данному адресу недоступна";
295  }
296  if (rw){
297  qemu_log("READ ONLY, ");
298  msg = "Память только для чтения";
299  }
300  if (user){
301  qemu_log("USER MODE, ");
302  msg = "Память только для пользователя";
303  }
304  if (reserved){
305  qemu_log("WRITING TO RESERVED BITS, ");
306  msg = "Попытка записи в зарез. биты";
307  }
308  if (id){
309  qemu_log("EIP error ");
310  msg = "Ошибка EIP";
311  }
312  qemu_log("at address (virtual) %x",fault_addr);
313 
314  // Prevent kmallocs (in bsod_screen()) in Page Fault
315  sod_screen_legacy(regs, "CRITICAL_ERROR_PF_PAGE_FAULT", msg, fault_addr);
316 }

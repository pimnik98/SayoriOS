465 int __com_init(uint16_t port) {
466  // qemu_log("[%x] Configurate",port);
467  outb(port + 1, 0x00); // Disable all interrupts
468  // qemu_log("[%x] Disable all interrupts",port);
469  outb(port + 3, 0x80); // Enable DLAB (set baud rate divisor)
470  // qemu_log("[%x] Enable DLAB",port);
471  outb(port + 0, 0x01); // Set divisor to 1 (lo byte) 115200 / divisor (1) = 115200 baud
472  // qemu_log("[%x] Set divisor to LO",port);
473  outb(port + 1, 0x00); // (hi byte)
474  // qemu_log("[%x] Set divisor to HI",port);
475  outb(port + 3, 0x03); // 8 bits, no parity, one stop bit
476  // qemu_log("[%x] 8 bits, no parity, one stop bit",port);
477  outb(port + 2, 0xC7); // Enable FIFO, clear them, with 14-byte threshold
478  // qemu_log("[%x] Enable FIFO, clear them, with 14-byte threshold",port);
479  outb(port + 4, 0x0B); // IRQs enabled, RTS/DSR set
480  // qemu_log("[%x] IRQs enabled, RTS/DSR set",port);
481  outb(port + 4, 0x1E); // Set in loopback mode, test the serial chip
482  // qemu_log("[%x] Set in loopback mode, test the serial chip",port);
483  outb(port + 0, 0xAE); // Test serial chip (send byte 0xAE and check if serial returns same byte)
484  // qemu_log("[%x] Test serial chip ",port);
485  // Check if serial is faulty (i.e: not same byte as sent)
486  // qemu_log("[%x] Check if serial is faulty",port);
487  if(inb(port + 0) != 0xAE) {
488  // qemu_log("An error occurred while configuring the com port %x",port);
489  return 1;
490  }
491  // qemu_log("[%x] If serial is not faulty set it in normal operation mode ",port);
492 
493  // If serial is not faulty set it in normal operation mode
494  // (not-loopback with IRQs enabled and OUT#1 and OUT#2 bits enabled)
495  outb(port + 4, 0x0F);
496  // __com_formatString(port,"COM Port %x configured successfully.",port);
497  return 0;
498 }

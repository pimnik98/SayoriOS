301 uint32_t pci_get_bar(uint8_t hdrtype, uint8_t bus, uint8_t slot, uint8_t func, uint8_t bar_number, uint8_t *bar_type) {
302  if ((hdrtype & ~0x80) == 0) {
303  uint8_t off = bar_number * 2;
304  uint16_t bar_low = pci_read_confspc_word(bus, slot, func, 0x10 + off);
305  uint16_t bar_high = pci_read_confspc_word(bus, slot, func, 0x10 + off + 1);
306  if ((bar_low & 1) == 0) {
307  if ((bar_low & ~0b110) == 0x00) // 32 бит. 64 бит нужно добавить!
308  {
309  uint32_t ret = (uint32_t) bar_low | (uint32_t) (bar_high << 16);
310  ret &= ~0b1111;
311  *bar_type = 0;
312  return ret;
313  }
314  }
315  if ((bar_low & 1) == 1) {
316  uint32_t ret = (uint32_t)bar_low | (uint32_t)(bar_high << 16);
317  ret &= ~0b11;
318  *bar_type = 1;
319  return ret;
320  }
321  }
322  return 0;
323 }

345 void pci_find_device(uint16_t vendor, uint16_t device, uint8_t *bus_ret, uint8_t *slot_ret, uint8_t *func_ret) {
346  for (uint32_t bus=0; bus<256; bus++)
347  for (uint32_t slot=0; slot<32; slot++)
348  for (uint32_t func=0; func<8; func++)
349  if (pci_get_device(bus, slot, func) == device && pci_get_vendor(bus, slot, func) == vendor) {
350  *bus_ret = bus;
351  *slot_ret = slot;
352  *func_ret = func;
353  return;
354  }
355  *bus_ret = *slot_ret = *func_ret = 0xFF;
356 }

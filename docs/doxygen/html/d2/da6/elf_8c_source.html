20 elf_sections_t* load_elf(const char* name){
21  /* Open ELF file */
22  FILE *file_elf = fopen(name, "r");
23 
24  if (file_elf->err) {
25  qemu_log("Failed to open ELF file: %s / %d", name, file_elf->err);
26  return NULL;
27  }
28 
29  size_t sz = 0;
30  /* Allocate ELF file structure */
31  elf_sections_t* elf = (elf_sections_t*) kmalloc(sizeof(elf_sections_t));
32  memset(elf, 0, sizeof(elf_sections_t));
33 
34  /* Read ELF header */
35  elf->elf_header = (Elf32_Ehdr*) kmalloc(sizeof(Elf32_Ehdr));
36  memset(elf->elf_header, 0, sizeof(Elf32_Ehdr));
37 
38  sz = fread_c(file_elf, 1, sizeof(Elf32_Ehdr), elf->elf_header);
39 
40  /* Read program header */
41  Elf32_Half proc_entries = elf->elf_header->e_phnum;
42  uint32_t proc_size = elf->elf_header->e_phentsize;
43 
44  elf->p_header = (Elf32_Phdr*)kmalloc(proc_size*proc_entries);
45  memset(elf->p_header, 0, proc_size*proc_entries);
46 
47  fseek(file_elf, elf->elf_header->e_phoff, SEEK_SET);
48 
49  sz = fread_c(file_elf, proc_entries, proc_size, elf->p_header);
50 
51  /* Read ELF sections */
52  Elf32_Half sec_entries = elf->elf_header->e_shnum;
53 
54  elf->section = (Elf32_Shdr*) kmalloc(sizeof(Elf32_Shdr)*sec_entries);
55  memset(elf->section, 0, sizeof(Elf32_Shdr)*sec_entries);
56 
57  fseek(file_elf, elf->elf_header->e_shoff, SEEK_SET);
58 
59  sz = fread_c(file_elf, sec_entries, sizeof(Elf32_Shdr), elf->section);
60 
61  elf->file = file_elf;
62 
63  (void)sz;
64 
65  return elf;
66 }

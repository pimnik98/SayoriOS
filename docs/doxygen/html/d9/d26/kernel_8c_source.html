126  mod_end = (uint32_t*) kmalloc(sizeof(uint32_t)*mods_count);
127 
128  qemu_log("[kModules] Found '%d' modules",mods_count);
129 
130  for (size_t i = 0; i < mods_count; i++){
131  mod_start[i] = *(uint32_t*)(multiboot->mods_addr + 8*i);
132  mod_end[i] = *(uint32_t*)(multiboot->mods_addr + 8*i + 4);

135 
136  mod_cmd[i] = kcalloc(strlen((char*)mod->cmdline) + 1, sizeof(char));
137  strcpy(mod_cmd[i], (char*)mod->cmdline);
138 
139  qemu_log("[kModules] Found module number `%d`. (Start: %x | End: %x) CMD: %s",i,mod_start[i],mod_end[i],mod_cmd[i]);

142  initrd_sefs(mod_start[i]);

144  }

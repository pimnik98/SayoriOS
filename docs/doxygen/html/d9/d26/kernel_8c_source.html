31 void kernelCMDHandler(char* cmd){
32  qemu_log("[kCMD] '%s'",cmd);
33  uint32_t kCMDc = str_cdsp(cmd," ");
34  char* out[128] = {0};
35  str_split(cmd,out," ");
36 
37  for(int i = 0; kCMDc >= i; i++){
38  qemu_log("[kCMD] %s",out[i]);
39  if (strcmpn(out[i],kCMD_BOOTSCREEN_MINIMAL)){
40  bootScreenChangeMode(1);
41  qemu_log("[kCMD] The minimum operating mode for BootScreen is selected.");
42  continue;
43  } else if (strcmpn(out[i],kCMD_NOPCISSAN)){
44  rtl8139_load = false;
45  changeScanToInit(false);
46  qemu_log("[kCMD] PCI devices will not be initialized at boot.");
47  continue;
48  } else if (strcmpn(out[i],kCMD_EXEC_TSHELL)){
49  autotshell = true;
50  qemu_log("[kCMD] After loading the kernel, TShell will automatically start.");
51  continue;
52  } else if (strcmpn(out[i],kCMD_DRIVER_BGA)){
53  bga_load = true;
54  qemu_log("[kCMD] The BGA driver will be loaded on kernel startup.");
55  continue;
56  } else if (strcmpn(out[i],kCMD_NO_DRIVER_RTL8139)){
57  rtl8139_load = false;
58  qemu_log("[kCMD] The Realtek RTL8139 driver will not be loaded on kernel startup.");
59  continue;
60  } else if (strcmpn(out[i],kCMD_NO_DRIVER_AC97)){
61  ac97_load = false;
62  qemu_log("[kCMD] The AC97 driver will not be loaded when the operating system starts.");
63  continue;
64  } else if (strcmpn(out[i],kCMD_BOOTSCREEN_LIGHT)){
65  bootScreenChangeTheme(1);
66  continue;
67  } else if (strcmpn(out[i],kCMD_BOOTSCREEN_DARK)){
68  bootScreenChangeTheme(0);
69  continue;
70  }
71  }
72 }

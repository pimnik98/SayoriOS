56 void ac97_init() {
57  // Find device
58  pci_find_device(AC97_VENDOR, AC97_DEVICE,
59  &ac97_busnum, &ac97_slot, &ac97_func);
60 
61  uint16_t devnum = pci_get_device(ac97_busnum, ac97_slot, ac97_func);
62 
63  qemu_log("AC'97 ID: %d (%x)", devnum, devnum);
64 
65  if(devnum == PCI_VENDOR_NO_DEVICE) {
66  qemu_log("AC'97 not connected!");
67  return;
68  }else{
69  qemu_log("Detected AC'97");
70  }
71 
72  // Enable IO Busmastering
73 
74  uint16_t command_register = pci_read_confspc_word(ac97_busnum, ac97_slot, ac97_func, 4);
75  qemu_log("Command register is: %x", command_register);
76 
77  command_register |= 0x05;
78 
79  qemu_log("Command register now is: %x", command_register);
80  pci_write(ac97_busnum, ac97_slot, ac97_func, 4, command_register);
81 
82  // Get NAM and NABM adresses for port i/o.
83 
84  native_audio_mixer = pci_read_confspc_word(ac97_busnum, ac97_slot, ac97_func, 0x10); // BAR0
85  native_audio_bus_master = pci_read_confspc_word(ac97_busnum, ac97_slot, ac97_func, 0x14); // BAR1
86 
87  // That's QEMU BUG?
88  native_audio_mixer--;
89  native_audio_bus_master--;
90 
91  qemu_log("NAM: %x; NABM: %x", native_audio_mixer, native_audio_bus_master);
92 
93  uint16_t extended_id = ins(native_audio_mixer + NAM_EXTENDED_ID);
94 
95  uint8_t rev = ((extended_id & (3 << 10)) >> 10) & 0b11;
96  qemu_log("Codec revision: (%d) %s", rev,
97  rev == 0 ? "r < 21" : (
98  rev == 1 ? "r22" : (
99  rev == 2 ? "r23"
100  : "Not supported"
101  )
102  )
103  );
104 
105  uint32_t gc = inl(native_audio_bus_master + NABM_GLOBAL_CONTROL);
106  qemu_log("Received global control: (%d) %x", gc, gc);
107  gc |= 2; // Cold reset
108 
109  uint32_t word = pci_read_confspc_word(ac97_busnum, ac97_slot, ac97_func, 0x3C); // All 0xF PCI register
110 
111  qemu_log("AC'97's IRQ is: %d", word & 0xff);
112 
113  outl(native_audio_bus_master + NABM_GLOBAL_CONTROL, gc);
114  outs(native_audio_mixer + NAM_RESET, 1);
115  qemu_log("Cold reset");
116 
117  // /*
118  // ac97_global_status_t status;
119  // ac97_global_status_t* statusptr = &status;
120 
121  uint32_t status = inl(native_audio_bus_master + NABM_GLOBAL_STATUS);
122  qemu_log("Status: %x", status);
123 
124  // qemu_log("Status: %d (%x)\n", status, status);
125  // qemu_log("Status Reserved: %d\n", status.reserved);
126  // qemu_log("Status Channels: %d\n", (status.channel==0?2:(status.channel==1?4:(status.channel==2?6:0))));
127  // qemu_log("Status Samples: %s\n", status.sample==1?"16 and 20 bits":"only 16 bits");
128 
129  // */
130 
131  uint16_t extended_audio = ins(native_audio_mixer + NAM_EXTENDED_AUDIO);
132  qemu_log("Status: %d", extended_audio);
133 
134  if((extended_id & 1) > 0) { // Check for variable sample rate
135  qemu_log("AC'97 supports variable sample rate!!!");
136  extended_audio |= 1;
137  ac97_varibale_sample_rate = true;
138  }
139 
140  outs(native_audio_mixer + NAM_EXTENDED_AUDIO, extended_audio);
141

143  ac97_set_pcm_sample_rate(44100);
144 
145  ac97_set_master_volume(0, 0, false);
146  ac97_set_pcm_volume(0, 0, false);
147 
148  qemu_log("Updated capabilities");
149 
150  ac97_initialized = true;

153  physpages_ac97 = kcalloc(currently_pages_count, sizeof(size_t));
154

156  physpages_ac97[i] = alloc_phys_pages(1);

167  }
168 
169  qemu_log("OK: Page count for 64 KB buffer is: %d", currently_pages_count);
170  qemu_log(" Physical pages position: %x", physpages_ac97);
171 
172  qemu_log("sizeof(BDL_ENTRY) = %d", sizeof(AC97_BDL_t));
173 
174  qemu_log("AC'97 initialized successfully!");

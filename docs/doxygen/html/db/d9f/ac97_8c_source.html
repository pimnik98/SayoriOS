106  native_audio_bus_master = pci_read_confspc_word(ac97_busnum, ac97_slot, ac97_func, 0x14); // BAR1

108  // That's QEMU BUG?

112  // uint8_t hdrtype = pci_get_hdr_type(ac97_busnum, ac97_slot, ac97_func);

114  // native_audio_bus_master = pci_get_bar(hdrtype, ac97_busnum, ac97_slot, ac97_func, 1, ac97_bar1_type); // BAR1
115 
116  qemu_log("NAM: %x; NABM: %x", native_audio_mixer, native_audio_bus_master);
117 
118  uint16_t extended_id = ins(native_audio_mixer + NAM_EXTENDED_ID);
119 
120  uint8_t rev = ((extended_id & (3 << 10)) >> 10) & 0b11;

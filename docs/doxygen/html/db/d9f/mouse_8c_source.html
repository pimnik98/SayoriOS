117 void mouse_parse_packet(const char *buf, uint8_t has_wheel, uint8_t has_5_buttons) {
118  mouse_flags_byte *mfb = (mouse_flags_byte*) (buf);
119  if (mfb->x_overflow || mfb->y_overflow || !mfb->always1) {
120  return;
121  }
122 
123  int offx = (int16_t) (0xff00 * mfb->x_sign) | buf[1];
124  int offy = (int16_t) (0xff00 * mfb->y_sign) | buf[2];
125  mouse_x += offx;
126  mouse_y -= offy;
127  mouse_b1 = mfb->left_button;
128  mouse_b2 = mfb->right_button;
129  mouse_b3 = mfb->middle_button;
130  ps2m_buffer.movement_x = offx;
131  ps2m_buffer.movement_y = offy;
132  ps2m_buffer.button_l = mouse_b1;
133  ps2m_buffer.button_r = mouse_b2;
134  ps2m_buffer.button_m = mouse_b3;
135 
136  if (has_wheel) {
137  mouse_wheel += (char) ((!!(buf[3] & 0x8)) * 0xf8 | (buf[3] & 0x7));
138  if (has_5_buttons) {
139  mouse_b4 = !!(buf[3] & 0x20);
140  // parse buttons 4-5 (byte 3, bits 4-5)
141  }
142  }
143 }

188 void mouse_handler(__attribute__((unused)) struct registers r) {
189  uint8_t status = inb(0x64);
190  if ((status & 1) == 0 || (status >> 5 & 1) == 0) {
191  return;
192  }
193 
194  static int recbyte = 0;
195  static char mousebuf[5];
196 
197  mousebuf[recbyte++] = inb(0x60);
198  if (recbyte == 3 /* + has_wheel */) {
199  recbyte = 0;
200 
201  static uint8_t drawn = 0;
202  if (drawn && show_mouse_cursor) {
203  mouse_erase();
204  }
205  drawn = 1;
206 
207  mouse_parse_packet(mousebuf, 0, 0);
208 
209  // Bounds
210  if (mouse_x < 0) mouse_x = 0;
211  if (mouse_y < 0) mouse_y = 0;
212  if (mouse_x > (int) (getWidthScreen())) mouse_x = getWidthScreen();
213  if (mouse_y > (int) (getHeightScreen())) mouse_y = getHeightScreen(); //-10;
214 
215  // Redraw the cursor
216  if(show_mouse_cursor) mouse_draw();
217  }
218 }

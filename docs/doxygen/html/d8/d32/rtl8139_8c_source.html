115 void rtl8139_setup_rcr() {
116  // AB - Accept Broadcast: Accept broadcast packets sent to mac ff:ff:ff:ff:ff:ff
117  // AM - Accept Multicast: Accept multicast packets.
118  // APM - Accept Physical Match: Accept packets send to NIC's MAC address.
119  // AAP - Accept All Packets. Accept all packets (run in promiscuous mode).
120 
121  // (1 << 7) is the WRAP bit, 0xf is AB+AM+APM+AAP
122 
123  outl(rtl8139_io_base + 0x44, 0xf | (1 << 7));
124 }

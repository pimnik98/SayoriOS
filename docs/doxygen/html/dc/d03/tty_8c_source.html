248 void tty_scroll() {
249  uint32_t num_rows = 1;
250  tty_pos_y -= tty_off_pos_h*num_rows;
251 
252  // Копируем строки сверху
253  uint8_t *read_ptr = (uint8_t*) back_framebuffer_addr + ((num_rows * tty_off_pos_h) * framebuffer_pitch);
254  uint8_t *write_ptr = (uint8_t*) back_framebuffer_addr;
255  uint32_t num_bytes = (framebuffer_pitch * VESA_HEIGHT) - (framebuffer_pitch * (num_rows * tty_off_pos_h));
256  memcpy(write_ptr, read_ptr, num_bytes);
257 
258  // Очистка строк
259  write_ptr = (uint8_t*) back_framebuffer_addr + (framebuffer_pitch * VESA_HEIGHT) - (framebuffer_pitch * (num_rows * tty_off_pos_h));
260  memset(write_ptr, 0, framebuffer_pitch * (num_rows * tty_off_pos_h));
261 
262  // Копируем буфферы
263  #if ENABLE_DOUBLE_BUFFERING==0
264  memcpy(framebuffer_addr, back_framebuffer_addr, framebuffer_size);

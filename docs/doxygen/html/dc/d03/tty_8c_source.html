261 void tty_scroll() {
262  uint32_t num_rows = 1;
263  tty_pos_y -= tty_off_pos_h*num_rows;
264 
265  // Копируем строки сверху
266  uint8_t *read_ptr = (uint8_t*) back_framebuffer_addr + ((num_rows * tty_off_pos_h) * framebuffer_pitch);
267  uint8_t *write_ptr = (uint8_t*) back_framebuffer_addr;
268  uint32_t num_bytes = (framebuffer_pitch * VESA_HEIGHT) - (framebuffer_pitch * (num_rows * tty_off_pos_h));
269  memcpy(write_ptr, read_ptr, num_bytes);
270 
271  // Очистка строк
272  write_ptr = (uint8_t*) back_framebuffer_addr + (framebuffer_pitch * VESA_HEIGHT) - (framebuffer_pitch * (num_rows * tty_off_pos_h));
273  memset(write_ptr, 0, framebuffer_pitch * (num_rows * tty_off_pos_h));
274 
275  // Копируем буфферы
276  #if ENABLE_DOUBLE_BUFFERING==0

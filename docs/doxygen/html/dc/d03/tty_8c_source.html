149  uint8_t *write_ptr = (uint8_t*) getFrameBufferAddr();
150 
151  uint32_t num_bytes = (getDisplayPitch() * VESA_HEIGHT) - (getDisplayPitch() * (num_rows * tty_off_pos_h));
152 
153  memcpy(write_ptr, read_ptr, num_bytes);
154 
155  // Очистка строк

157  memset(write_ptr, 0, getDisplayPitch() * (num_rows * tty_off_pos_h));

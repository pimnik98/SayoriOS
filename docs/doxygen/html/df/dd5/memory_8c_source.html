41 void switch_page_mode(void){
42  qemu_log("Reached switch page mode");
43  size_t vaddr = 0;
44  uint32_t frame = 0;
45  physaddr_t paddr = 0;
46  uint32_t table_idx = 0;
47  uint32_t page_idx = 0;
48  uint32_t cr0;
49  uint32_t table_flags = 0;
50  uint32_t page_flags = 0;
51 
52  // Create kernel page directory
53  uint32_t* kernel_dir = (uint32_t*)kernel_page_dir;
54 
55  qemu_log("Clearing kernel directory");
56 
57  // Clear kernel directory
58  memset(kernel_dir, 0, PAGE_SIZE);
59 
60  qemu_log("Ok");
61 
62  // Map all memory pages for kernel
63  for (vaddr = 0; vaddr < (KERNEL_BASE + KERNEL_SIZE); vaddr += PAGE_SIZE){
64  // Translate virtual address
65  // qemu_log("Mapping vaddr: %x; paddr: %x", vaddr, paddr);
66 
67  // !!!!!!!!
68  // Translation of a virtual address into a physical address first
69  // involves dividing the virtual address into three parts:
70  //
71  // the most significant 10 bits (bits 22-31) specify the index of the page directory entry
72  // the next 10 bits (bits 12-21) specify the index of the page table entry,
73  // and the least significant 12 bits (bits 0-11) specify the page offset.
74  // !!!!!!!!
75 
76  frame = vaddr >> PAGE_OFFSET_BITS; // The address
77  table_idx = frame >> PAGE_TABLE_INDEX_BITS; // Page directory entry (contain table)
78  page_idx = frame & PAGE_TABLE_INDEX_MASK; // Page
79 
80  // Set table flags
81  table_flags = PAGE_PRESENT | PAGE_WRITEABLE;
82 
83  // Create page table
84  *(kernel_dir + table_idx) = (uint32_t)(GET_PT() + table_idx * 0x400) | table_flags;
85 
86  // Set page flags
87  page_flags = PAGE_PRESENT | PAGE_WRITEABLE;
88 
89  // Create page
90  *(GET_PT() + table_idx * 0x400 + page_idx) = paddr | page_flags;
91 
92  paddr += PAGE_SIZE;
93  }
94 
95  qemu_log("Ok, loading kernel directory address to cr3 register: %x", kernel_page_dir);
96 
97  // Load kernel directory address to CR3
98  asm volatile ("mov %0, %%cr3"::"r"(kernel_page_dir));
99 
100  qemu_log("Load ok");
101 
102  // Switch to paging mode
103  asm volatile ("mov %%cr0, %0":"=r"(cr0));
104  cr0 |= 0x80000001;
105  asm volatile ("mov %0, %%cr0"::"r"(cr0));
106 
107  qemu_log("Switched successfully");
108 }

218  tmp_block->size += new_count;
219  }
220 
221  break;
222  }
223

225  if (cur_block == base + (count << PAGE_OFFSET_BITS)){
226  /* Remember old size and next and previous */

228  physaddr_t next = tmp_block->next;
229  physaddr_t prev = tmp_block->prev;
230

242

244  tmp_block = get_free_block(base);
245 
246  tmp_block->next = next;

248  tmp_block->size += old_count;
249 
250  break;
251  }
252

259  tmp_block->next = base;

270  if (tmp_block->next == free_phys_memory_pointer){
271  tmp_block->prev = base;
272 
273  tmp_block->next = base;
274 
275  tmp_block = get_free_block(base);
276 
277  tmp_block->prev = cur_block;
278  tmp_block->next = cur_block;
279  tmp_block->size = count;
280 
281  break;
282  }

285 
286  } while ( cur_block != free_phys_memory_pointer );
287 
288  if (base < free_phys_memory_pointer){
289  free_phys_memory_pointer = base;
290  }
291  }

303 physaddr_t alloc_phys_pages(size_t count){
304  physaddr_t result = -1;
305  physmemory_pages_block_t* tmp_block;
306 
307  mutex_get(&phys_memory_mutex, true);
308 
309  /* Number of free pages less than count - error!*/
310  if (free_pages_count < count)
311  return -1;
312 
313  if (free_phys_memory_pointer != -1){
314  physaddr_t cur_block = free_phys_memory_pointer;
315 
316  do{
317  /* Get current free block*/
318  tmp_block = get_free_block(cur_block);
319 
320  /* If block and count have same size*/
321  if ( tmp_block->size == count ){
322  /* Remember next and previous */
323  physaddr_t next = tmp_block->next;
324  physaddr_t prev = tmp_block->prev;
325 
326  /* Get next block */
327  tmp_block = get_free_block(next);
328 
329  /* For next block - previous is previous for allocated block */
330  tmp_block->prev = prev;
331 
332  /* Get previous block*/
333  tmp_block = get_free_block(prev);
334 
335  /* For next block - next is next for allocated block */
336  tmp_block->next = next;
337 
338  /* If it was penultimate free block */
339  if (cur_block == free_phys_memory_pointer){
340  /* Check next */
341  free_phys_memory_pointer = next;
342 
343  /* If next block is current - we have no free memory */
344  if (cur_block == free_phys_memory_pointer)
345  {
346  free_phys_memory_pointer = -1;
347  }
348  }
349 
350  /* Result address of allocated block */
351  result = cur_block;
352  break;
353  }
354 
355  /* If current free block more than count of pages */
356  if ( tmp_block->size > count ){
357  /* Reduce size of current block */
358  tmp_block->size -= count;
359 
360  /* Result address - current block's address + size of allocated block*/
361  result = cur_block + (tmp_block->size << PAGE_OFFSET_BITS);
362  break;
363  }
364 
365  /* Get next free block */
366  cur_block = tmp_block->next;
367 
368  } while (cur_block != free_phys_memory_pointer);
369 
370  /* If we have correct address */
371  if (result != -1){
372  /* Reduce free page count */
373  free_pages_count -= count;
374  }
375  }
376 
377  mutex_release(&phys_memory_mutex);
378 
379  return result;
380 }

687  mutex_get(&thismutex, true);

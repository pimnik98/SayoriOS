95 void devmgr_cli(uint32_t c,char* v[]){
96  char* cmd = kheap_malloc(sizeof(char)*256);
97  cmd = v[1];
98  uint32_t param1 = atoi(v[2]);
99  uint32_t param2 = atoi(v[3]);
100  qemu_log("[Test KB] %d %s %d %d",c,cmd,param1,param2);
101  tty_printf("Менеджер устройств v%d.%d.%d\n",DEVMGR_VERSION_MAJOR, DEVMGR_VERSION_MINOR, DEVMGR_VERSION_PATCH);
102  if (strcmpn(cmd,"help") || c == 0){
103  devmgr_cli_help();
104  return;
105  }
106  if (strcmpn(cmd,"reg")){
107  if (c != 3){
108  tty_setcolor(COLOR_ERROR);
109  tty_printf("Необходимо использовать 4 параметра!\n");
110  tty_printf("Пример:\n");
111  tty_printf("\t devmgr reg 32902 10200 - Подключит к системе уст-во с поставщиком 0x8086 и устройством 0x27D8\n");
112  tty_printf("\t devmgr reg 4660 4369 - Подключит к системе уст-во с поставщиком 0x1234 и устройством 0x1111\n");
113  tty_printf("\t devmgr reg 4660 4370 - Подключит к системе уст-во с поставщиком 0x1234 и устройством 0x1112\n");
114  return;
115  }
116  devmgr_cli_reg(param1,param2);
117  return;
118  }
119  if (strcmpn(cmd,"info")){
120  if (c != 2){
121  tty_setcolor(COLOR_ERROR);
122  tty_printf("Необходимо использовать 3 параметра!\n");
123  tty_printf("Пример:\n");
124  tty_printf("\t devmgr info 0\n");
125  return;
126  }
127  devmgr_cli_info(param1);
128  return;
129  }
130  if (strcmpn(cmd,"vendor")){
131  if (c != 2){
132  tty_setcolor(COLOR_ERROR);
133  tty_printf("Необходимо использовать 3 параметра!\n");
134  tty_printf("Пример:\n");
135  tty_printf("\t devmgr vendor 32902\n");
136  return;
137  }
138  devmgr_cli_vendor(param1);
139  return;
140  }
141  devmgr_cli_help();
142  tty_setcolor(COLOR_ERROR);
143  tty_printf("Неправильно заданы параметры.\n");
144  tty_printf("\t devmgr '%s' '%d' '%d' '%d'",cmd,param1,param2,(strcmp(cmd,"reg")));
145  return;
146 }

168 void rgba_blend(uint8_t result[4], const uint8_t fg[4], const uint8_t bg[4])
169 {
170  uint32_t alpha = fg[3] + 1;
171  uint32_t inv_alpha = 256 - fg[3];
172 
173  result[0] = (uint8_t)((alpha * fg[0] + inv_alpha * bg[0]) >> 8);
